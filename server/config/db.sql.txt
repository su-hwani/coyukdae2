DROP TABLE IF EXISTS IMAGEBUTTON;
DROP TABLE IF EXISTS SELECTEDIMAGE;

CREATE TABLE IMAGEBUTTON (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    IMAGEURL TEXT DEFAULT NULL,
    IMAGENAME VARCHAR(255) DEFAULT NULL
);

CREATE TABLE SELECTEDIMAGE (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    SESSIONID TEXT DEFAULT NULL,
    ROUND_32 TEXT DEFAULT NULL,
    ROUND_16 TEXT DEFAULT NULL,
    ROUND_8 TEXT DEFAULT NULL,
    ROUND_4 TEXT DEFAULT NULL,
    ROUND_2 TEXT DEFAULT NULL
);
